-- ПРОБЕЛЫ В НАЗВАНИЯХ ПОЛЕЙ ТАБЛИЦЫ ЗАМЕНИЛА НА _

--1
SELECT *  

FROM  ВОДИТЕЛИ 

WHERE СТАЖ NOT IN (3,5,6,10) AND МЕСТО_ПРОПИСКИ=’еее’ 

--2
SELECT №_ПАСПОРТА 

FROM  ВОДИТЕЛИ В  LEFT JOIN  ДИСПЕТЧЕРСКАЯ Д ON  В.КОД_ВОДИТЕЛЯ = Д.КОД_ВОДИТЕЛЯ 

WHERE Д.КОД_ВОДИТЕЛЯ IS NULL 

--3

SELECT А.РЕГИСТРАЦИОННЫЙ_№_АВТО, Д.ВРЕМЯ_ПРИБЫТИЯ, Д.ВРЕМЯ_ОТБЫТИЯ, M.ПУНКТ_НАЗНАЧЕНИЯ 

FROM AВТО A JOIN  ДИСПЕТЧЕРСКАЯ Д ON А.КОД_АВТО = Д.КОД_АВТО  JOIN  МАРШРУТ M ON Д.ПУТЕВКА=М.КОД_МАРШРУТА 

WHERE А.ГОД_ВЫПУСКА_АВТО=’2015’
